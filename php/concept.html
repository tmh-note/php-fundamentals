PHP Operator Example

Arithmetic operators
                
                    echo 5 + 3;     // prints 8
                    echo 5 - 3;     // prints 2
                    echo 5 * 3;     // prints 15
                    echo 6 / 3;     // prints 2
                    echo 6 % 3;     // prints 0
Assignment operators
 
                
                    $a = 3;
                    $a += 5; // sets $a to 8, as if we had said: $a = $a + 5;
                    $b = "Hello ";
                    $b .= "There!"; // sets $b to "Hello There!", just like $b = $b . "There!";
Comparison operators
 
                
                    var_dump(1 == 1); // true
                    var_dump(1 != 2); // false
                    var_dump(1 > 2); // false
                    var_dump(1 < 2); // true
                    var_dump(1 >= 2); // false
                    var_dump(1 <= 2); // true
Increment/Decrement operators
 
                
                    // increment
                    $i = 1;
                    echo $i; // 1
                    $i += 1;
                    echo $i; // 2
                    $i++;
                    echo $i; // 3
                    echo ++$i; // 4

                    // decrement
                    $i = 5;
                    echo $i; // 5
                    $i -= 1;
                    echo $i; // 4
                    $i--;
                    echo $i; // 3
                    echo --$i;  //2
Logical operators
 
                
                    var_dump(false and true); // false
                    var_dump(false or true); // true
                    var_dump(false && true); // false
                    var_dump(false || true); // true
String operators
 
                
                    $a = "Hello ";
                    $b = $a . "World!"; // now $b contains "Hello World!"
                    
                    $a = "Hello ";
                    $a .= "World!";     // now $a contains "Hello World!"
Session
                
                    session_start();

                    $_SESSION['msg'] = 'Hello PHP';
                    echo $_SESSION['msg'];

                    // remove session variables
                    unset($_SESSION['msg']);

                    // remove all session variables
                    session_unset();

                    // destroy the session
                    session_destroy();
Cookie
                
                    Setting new cookie
                    ============================= 
                    setcookie("msg","Hello PHP",time()+60); // 1 minute
                        
                    Getting Cookie
                    =============================
                    echo $_COOKIE["msg"];
                    
                    Updating Cookie
                    =============================
                    setcookie("msg","Hello World");
                    echo $_COOKIE["msg"];
                    
                    Deleting Cookie
                    ==============================
                    unset($_COOKIE["msg"]);
                    /*Or*/
                    setcookie("yourcookie","yourvalue",time()-1);
                    /*it expired so it's deleted*/
